import java.util.ArrayList;
import java.util.List;

public class ConfigManager {

    // 假设这是一个存储接口源的列表
    private List<Source> sources = new ArrayList<>();

    // 构造函数初始化接口源
    public ConfigManager() {
        // 只保留文才弹幕和低端纯净两个接口
        sources.add(new Source("文才弹幕", "http://example.com/danmu_source", true));
        sources.add(new Source("低端纯净", "http://example.com/low_quality_source", true));
    }

    // 获取所有启用的接口源
    public List<Source> getEnabledSources() {
        List<Source> enabledSources = new ArrayList<>();
        for (Source source : sources) {
            if (source.isEnabled()) {
                enabledSources.add(source);
            }
        }
        return enabledSources;
    }

    // 打印当前配置的所有接口源
    public void printSources() {
        for (Source source : sources) {
            System.out.println(source);
        }
    }

    // 添加接口源
    public void addSource(String name, String url, boolean enabled) {
        sources.add(new Source(name, url, enabled));
    }

    // 移除接口源
    public void removeSource(String name) {
        sources.removeIf(source -> source.getName().equals(name));
    }
}
